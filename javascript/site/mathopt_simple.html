MathOpt - linear programming sample

import { MathOpt, initMathOpt } from 'or-tools-wasm';

MathOpt.setWorkerBridgeEnabled(true);
await initMathOpt();

const model = MathOpt.Model('Linear programming example');
const x0 = model.addVariable({ lowerBound: 0, name: 'x0' });
const x1 = model.addVariable({ lowerBound: 0, name: 'x1' });
const x2 = model.addVariable({ lowerBound: 0, name: 'x2' });

// Same constraints as linear_programming.py.
model.addLinearConstraint({ upperBound: 600, terms: [
  { variable: x0, coefficient: 10 }, { variable: x1, coefficient: 4 }, { variable: x2, coefficient: 5 },
]});
model.addLinearConstraint({ upperBound: 300, terms: [
  { variable: x0, coefficient: 2 }, { variable: x1, coefficient: 2 }, { variable: x2, coefficient: 6 },
]});
model.addLinearConstraint({ upperBound: 100, terms: [x0, x1, x2].map((variable) => ({ variable, coefficient: 1 })) });
model.maximize([
  { variable: x0, coefficient: 10 }, { variable: x1, coefficient: 6 }, { variable: x2, coefficient: 4 },
]);

const result = await MathOpt.solve(model, {
  solverType: MathOpt.SolverType.GLOP,
  threads: 4,
});
console.log(result.objectiveValue, result.variableValues);

Direct port of ortools/math_opt/samples/python/linear_programming.py. GLOP maximizes 10x0 + 6x1 + 4x2 under three resource constraints.
