CLP - transportation LP

Problem: decide how much each plant should ship to each warehouse so all demand is met at the lowest total shipping cost.

Show code 
import { initMPSolver, MPSolver, setWorkerBridgeEnabled } from 'or-tools-wasm';

setWorkerBridgeEnabled(true);
await initMPSolver();
const solver = MPSolver.CreateSolver('CLP');
if (!solver) throw new Error('CLP is unavailable');

const plants = [
  { name: 'Detroit', supply: 350 },
  { name: 'Denver', supply: 600 },
  { name: 'Austin', supply: 500 },
];
const warehouses = [
  { name: 'Boston', demand: 325 },
  { name: 'Atlanta', demand: 300 },
  { name: 'Chicago', demand: 275 },
  { name: 'Seattle', demand: 250 },
];
const costs = [
  [2.7, 1.8, 1.4, 2.9],
  [2.2, 2.5, 1.7, 1.1],
  [1.9, 1.6, 2.3, 2.4],
];

const ship = plants.map((plant, plantId) =>
  warehouses.map((warehouse, warehouseId) =>
    solver.NumVar(0, solver.infinity(), `ship_${plant.name}_${warehouse.name}`),
  ),
);

plants.forEach((plant, plantId) => {
  const ct = solver.Constraint(0, plant.supply);
  warehouses.forEach((_, warehouseId) => ct.SetCoefficient(ship[plantId][warehouseId], 1));
});
warehouses.forEach((warehouse, warehouseId) => {
  const ct = solver.Constraint(warehouse.demand, warehouse.demand);
  plants.forEach((_, plantId) => ct.SetCoefficient(ship[plantId][warehouseId], 1));
});

const objective = solver.Objective();
plants.forEach((_, plantId) => {
  warehouses.forEach((_, warehouseId) => {
    objective.SetCoefficient(ship[plantId][warehouseId], costs[plantId][warehouseId]);
  });
});
objective.SetMinimization();

await solver.Solve();
console.log(objective.Value());

A continuous transportation LP: choose shipment quantities from plants to warehouses while meeting demand at minimum cost with the COIN-OR CLP backend.
