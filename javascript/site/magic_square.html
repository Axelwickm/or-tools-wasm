CP-SAT - Magic Square

import { CpSat, type SatParameters } from 'or-tools-wasm';

const model = await CpSat.createModel({
  name: 'magic_square_3',
  variables: Array.from({ length: 9 }, (_, i) => ({ name: `cell_${i}`, domain: [1, 9] })),
  constraints: [
    { all_diff: { exprs: Array.from({ length: 9 }, (_, i) => ({ vars: [i], coeffs: [1] })) } },
    // Add row, column, and diagonal linear constraints with domain [15, 15].
  ],
});

CpSat.setWorkerBridgeEnabled(true);
const params: SatParameters = { numWorkers: 4, logSearchProgress: true };
const result = await CpSat.solve(model, params);
console.log(result.response?.solution);
