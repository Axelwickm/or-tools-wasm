import { CpSat, type SatParameters } from 'or-tools-wasm';

const modelJson = JSON.parse(textarea.value);
const params: SatParameters = {
  ...JSON.parse(paramsTextarea.value),
  numWorkers: 4,
};

CpSat.setWorkerBridgeEnabled(true);
const model = await CpSat.createModel(modelJson);
const validation = await CpSat.validate(model);
if (!validation.ok) throw new Error(validation.message);

const result = await CpSat.solve(model, params);
console.log(result.response);
