Network Flow - max flow

Problem: send as much flow as possible from a source node to a sink node through arcs with limited capacities.

Show code 
import { initNetworkFlow, SimpleMaxFlow, setWorkerBridgeEnabled } from 'or-tools-wasm/network-flow';

setWorkerBridgeEnabled(true);
await initNetworkFlow();

const startNodes = [0, 0, 0, 1, 1, 2, 2, 3, 3];
const endNodes = [1, 2, 3, 2, 4, 3, 4, 2, 4];
const capacities = [20, 30, 10, 40, 30, 10, 20, 5, 20];

const maxFlow = new SimpleMaxFlow();
const allArcs = maxFlow.add_arcs_with_capacity(startNodes, endNodes, capacities);
const status = await maxFlow.solve(0, 4);

if (status === SimpleMaxFlow.OPTIMAL) {
  console.log(maxFlow.optimal_flow());
  console.log(allArcs.map((arc) => ({
    tail: maxFlow.tail(arc),
    head: maxFlow.head(arc),
    capacity: maxFlow.capacity(arc),
    flow: maxFlow.flow(arc),
  })));
  console.log(maxFlow.get_source_side_min_cut(), maxFlow.get_sink_side_min_cut());
}
